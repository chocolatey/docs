**EXAMPLE 2**

$stepParams = @{

    Step = 'Gather Info'
    TargetGroup = 'US-East servers'

                [pscustomobject]@{
                    Name = $data.caption

                }
    }
}
Set-CCMDeploymentStep @stepParams
